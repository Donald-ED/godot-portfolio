state_machine.gd 
class_name StateMachine extends Node

@export var initial_state: State
var current_state: State
var states: Dictionary = {}

func _ready() -> void:
    for child in get_children():
        if child is State:
            states[child.name.to_lower()] = child
            child.transitioned.connect(_on_state_transitioned)
    if initial_state:
        initial_state.enter()
        current_state = initial_state

func _on_state_transitioned(state: State, new_state_name: String) -> void:
    if state != current_state:
        return
    var new_state = states.get(new_state_name.to_lower())
    if not new_state:
        return
    current_state.exit()
    new_state.enter()
    current_state = new_state

static func axial_to_pixel(q: int, r: int) -> Vector2:
    var x = TILE_SIZE.x * (3.0/2.0 * q)
    var y = TILE_SIZE.y * (sqrt(3.0)/2.0 * q + sqrt(3.0) * r)
    return Vector2(x, y)

static func distance(a: Vector2i, b: Vector2i) -> int:
    return (abs(a.x - b.x) + abs(a.x + a.y - b.x - b.y) + abs(a.y - b.y)) / 2

func host_game() -> void:
    peer = ENetMultiplayerPeer.new()
    peer.create_server(PORT, MAX_PLAYERS)
    multiplayer.multiplayer_peer = peer

    multiplayer.peer_disconnected.connect(_on_peer_disconnected)

dialog_runner.gd 
func _process_entry() -> void:
    var entry = dialog_tree.get_entry(current_entry_id)
    if not entry:
        dialog_ended.emit()
        return
    match entry.type:
        "dialog":
            var text = _interpolate(entry.text)
            line_displayed.emit(entry.speaker, text)
        "condition":
            var result = _evaluate(entry.condition)
            var next = dialog_tree.get_next_entries(current_entry_id)
            for conn in next:
                if (result and conn.from_port == 0) or (not result and conn.from_port == 1):
                    current_entry_id = conn.to_id
                    _process_entry()
                    return
        "choice":
            choices_presented.emit(entry.choices)
